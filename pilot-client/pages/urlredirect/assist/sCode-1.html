    url = request.args['url']


    # 判断url是否以'http://'或'https://'开头
    if url.startswith('http://') or url.startswith('https://'):
        # 如果是，则直接重定向到该url
        return redirect(url)
    else:
        # 如果不是，则返回公共错误回调函数responses.callback_public_urlerr1
        return responses.callback_public_urlerr1
