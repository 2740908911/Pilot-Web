def download():
    # 从请求参数中获取文件路径
    file_path = request.args['file']
    try:
        # 拼接文件路径
        file = "download/" + file_path
        # 发送文件作为附件
        return send_file(file, as_attachment=True)
    except:
        # 发生异常时返回文件错误响应
        return responses.callback_public_fileerr1
