def Login():
    # 从请求中获取用户名和密码
    username = request.form['username']
    password = request.form['password']


    # 构建SQL查询语句
    sql = "SELECT USERNAME FROM Dbo.[USER] WHERE (USERNAME = '" + username + "' AND PASSWORD = '" + hashlib.md5(password.encode("utf-8")).hexdigest() + "')"


    try:
        # 执行SQL查询
        user = query_mssql(sql,)


        # 如果查询结果存在
        if user:
            # 返回登录成功的回调函数，并携带用户名和令牌
            return responses.update_callback(responses.callback_public_loginsucc, {'username': user[0][0], 'token': create_token(user[0][0])})
        else:
            # 返回登录失败的回调函数
            return responses.callback_public_loginerr1


    except Exception as e:
        # 如果发生异常，返回SQL错误的回调函数，并携带异常信息
        return responses.update_callback(responses.callback_public_sqlerror, {'msg': str(e)})


def getUserinfo():
    # 从请求头中获取token
    token = request.headers['Authorization']
    # 从请求参数中获取用户名
    username = request.args['n']


    # 将token中的第二部分进行base64解码，并获取account信息
    account = base64.b64decode(add_base64_padding(token.split('.')[1])).decode().split('"account":"')[1].split('"')[0]


    # 验证token是否有效
    if verify_token(token,account):
        # 构造查询用户信息的SQL语句
        sql = "SELECT UID FROM Dbo.[USER] WHERE USERNAME = '" + username + "'"
        try:
            # 执行SQL查询，判断用户是否存在
            if query_mssql(sql,)[0][0] == 1:
                # 如果用户存在，则查询用户详细信息
                res = query_mssql("SELECT USERNAME,PHONE,QQ,PID,UID FROM Dbo.[USER]")
                # 返回用户信息
                return responses.update_callback(responses.callback_public_getinfo, {"msg": transform_data(res)}) 
            else:
                # 用户不存在，返回认证错误回调
