function doLogin(){
    // 初始化验证码验证标志为false
    let verify = false;


    // 获取用户名输入框的值
    let username = $("#username").val();
    // 获取密码输入框的值
    let password = $("#password").val();


    // 获取验证码输入框的值，并转换为小写
    let inputCode = $("#code").val().toLowerCase();
    // 获取生成的验证码，并转换为小写
    let generatedCode = currentCode.toLowerCase();


    // 如果用户名、密码或验证码为空，则显示登录失败提示，并返回
    if (username.length <= 0 || password.length <= 0 || inputCode.length <= 0) {
      $("#notice")[0].innerHTML = generateNote("登录失败：缺少参数");
      return;
    }


    // 如果输入的验证码与生成的验证码相同，则验证通过，设置验证标志为true
    if (inputCode === generatedCode) {
      verify = true; 
    } else {
      // 如果验证码错误，则显示验证码错误提示，并返回
      $("#notice").html(generateNote("验证码错误！"));
      return;
    }


    // 发送登录请求
    $.post({
      url: `http:/ip:port/api/vc/l1/login`,
      data: {
        // 将用户名、密码和验证标志作为请求参数
        username: username,
        password: password,
        verify: verify
      },
      dataType: "json",
      // 请求成功的回调函数
      success(resp){
        // 如果服务器返回状态码为200，则显示登录成功提示，并禁用登录按钮
        if(resp["status"] === "200"){
          $("#notice")[0].innerHTML = generateNote("登录成功！欢迎, " + resp["username"]);
          $("#loginButton").prop('disabled', true);
        // 如果服务器返回状态码为405，则显示登录失败提示
        } else if(resp["status"] === "405"){
          $("#notice")[0].innerHTML = generateNote("登录失败：" + resp["msg"]);
        // 如果服务器返回状态码为401，则显示登录失败提示
        } else if(resp["status"] === "401"){
          $("#notice")[0].innerHTML = generateNote("登录失败：" + resp["msg"]);
        }
      }
    })
  }
