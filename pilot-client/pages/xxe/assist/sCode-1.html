def parse_xml():
    # 从表单中获取XML数据
    xml_data = request.form['xml']
    try:
        # 创建XML解析器，设置加载DTD和解析实体
        parser = etree.XMLParser(load_dtd=True, resolve_entities=True)
        # 从字符串中解析XML数据
        tree = etree.fromstring(xml_data, parser=parser)


        # 初始化content变量
        content = ''
        # 遍历XML树的所有元素
        for elem in tree.iter():
            # 将元素的标签和文本内容添加到content中
            content += elem.tag + ': ' + (elem.text or '') + '\n'


        # 调用responses.update_callback方法，并传入回调函数和包含解析内容的字典
        return responses.update_callback(responses.callback_public_getinfo, {"msg": content})
    # 捕获XML语法错误异常
    except etree.XMLSyntaxError as e:
        # 调用responses.update_callback方法，并传入回调函数和包含错误信息的字典
        return responses.update_callback(responses.callback_public_getinfo, {"msg": str(e)})
