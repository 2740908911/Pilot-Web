def fetch():
    # 获取url参数
    url = request.args['url']


    # 如果url以'file://'开头
    if url.startswith('file://'):
        try:
            # 提取文件路径
            filepath = url[7:]
            # 打开文件并读取内容
            with open(filepath, 'r', encoding="utf-8") as file:
                # 返回文件内容
                return file.read()
        except Exception as e:
            # 打印异常信息
            print(e)
            # 返回错误回调
            return responses.callback_public_urlerr1
    else:
        try:
            # 发送GET请求
            res = requests.get(url)
            # 设置返回结果的编码为utf-8
            res.encoding = 'utf-8'
            # 返回请求结果
            return res.text
        except Exception as e:
            # 返回错误回调
            return responses.callback_public_urlerr1
